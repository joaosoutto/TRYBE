function verificaPalindrome(palavra) {
    for ( let i = 0; i < palavra.length; i+=1) {
      if (palavra[i] !== palavra[palavra.length - 1 - i]) {
        return false;
      }
    }
    return true;
  }

  function highestI (arr) {
    let highestValue = arr[0];
    let index = 0;
    for ( let i = 0; i < arr.length; i+=1 ) {
      if ( highestValue < arr[i]) {
        highestValue = arr[i];
        index = i;
      }
    }
    return index;
  }
